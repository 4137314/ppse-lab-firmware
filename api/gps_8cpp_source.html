18bool gpsInit() {
19 Serial.println("=== GPS Init ===");
20
21 // Led alive init
22 pinMode(LED_ALIVE, OUTPUT);
23 digitalWrite(LED_ALIVE, LOW); // LED alive off
24 Serial.print("Led alive pin: ");
25 Serial.print(LED_ALIVE);
26 Serial.println(" inizializzato come output.");
27
28 // Enable GPS module
29 pinMode(GPS_EN_PIN, OUTPUT);
30 digitalWrite(GPS_EN_PIN, LOW); // bring down GPS_EN signal to enable (pmos)
31 Serial.print("GPS enable pin: ");
32 Serial.print(GPS_EN_PIN);
33 Serial.println(" alimentato.");
34
35 // Enable input for Pulse signal from gps
36 pinMode(GPS_PPS_PIN, INPUT);
37 Serial.print("Gps PPS pin: ");
38 Serial.print(GPS_PPS_PIN);
39 Serial.println(" inizializzato come input.");
40
41 // UART1 for the GPS. Serial2 maps to UART1 in Arduino framework!
42 Serial2.setTX(UART1_TX_PIN);
43 Serial2.setRX(UART1_RX_PIN);
44 Serial2.begin(GPSBAUD); // start uart1
45 Serial.println("UART1 started on Serial2, listening for GPS data...");
46
47 // Wait at least 10 sec for a GPS string.
48 int i;
49 for (i = 0; i < 1000 && (Serial2.available() == 0); ++i)
50 delay(1);
51
52 if (i == 1000) {
53 Serial.println("ERROR: Canot start the GPS module");
54 return false;
55 }
56 Serial.println("OK!");
57
58 return true;
59}

104 if (minmea_parse_gst(frame, nmea_message->c_str()))
105 print_NMEA_gst(frame);
106 else
107 (state = false);
108 } break;
109
110 case MINMEA_SENTENCE_GSV: {
111 struct minmea_sentence_gsv *frame = &(nmea_ptr->parsed_gsv);

153bool gpsAcquire(enum minmea_sentence_id sentence_type,
154 struct parsed_nmea *nmea_ptr) {
155 String nmea_message;
156
157 digitalWrite(LED_ALIVE, HIGH);
158
159 for (int i = 0; i < GPS_ACQUIRE_MAX_TRIES; ++i) {
160 for (int c = 0; Serial2.available() == 0; ++c)
161 delay(1);
162
163 nmea_message = Serial2.readStringUntil('\n');
164 digitalWrite(LED_ALIVE, LOW);
165
166 if (minmea_sentence_id(nmea_message.c_str(), true) == sentence_type) {
167 if (nmea_gps_parse(&nmea_message, nmea_ptr))
168 return true;
169 }
170 }
171
172#if DEBUG
173 Serial.println("WARNING in gpsAcquire: max tries reached before correctly "
174 "acquiring the requested gps message\n");
175#endif
176 return false;
177}

209bool GPS_sync(struct parsed_nmea *nmea_ptr, bool force) {
210 // retain value across function calls
211 static unsigned long lastGPSsync = 0, lastGPSwrite = 0;
212
213 if ((millis() - lastGPSsync > GPS_SYNC_TIMEOUT_MSEC) || force) {
214 if (GetPosition_and_Satellites(nmea_ptr) && GetDate_and_Time(nmea_ptr)) {
215 lastGPSsync = millis();
216 if (((millis() - lastGPSwrite) > GPS_WRITE_FREQ) ||
217 force) // Write only at a certain frequency otherwise flash will wear
218 // out
219 {
220 if (save_gps_last(nmea_ptr))
221 lastGPSwrite = millis();
222 else
223 return false;
224 }
225 } else // It's only a warning so return true at the end
226 {
227#if DEBUG
228 Serial.println("WARNING in GPS_sync: failed to get data");
229#endif
230 }
231 }
232 return true;
233}

242bool save_gps_last(struct parsed_nmea *nmea_ptr) {
243 if (inPrinting || driveConnected || updated)
244 return false;
245
246 struct minmea_sentence_gga *gga = &(nmea_ptr->parsed_gga);
247 if (gga->fix_quality <= 0)
248 return false;
249 char line[96];
250 float lat = minmea_tocoord(&(gga->latitude));
251 float lon = minmea_tocoord(&(gga->longitude));
252
253 inPrinting = true;
254 FILE *f = fopen(GPS_LAST_PATH, "w"); // sovrascrive sempre
255 if (!f) {
256 inPrinting = false;
257 return false;
258 }

260 fwrite(line, 1, (size_t)n, f);
261 fclose(f);
262 inPrinting = false;
263 return true;

273bool save_gps_log20(struct parsed_nmea *nmea_ptr) {
274 if (inPrinting || driveConnected || updated)
275 return false;
276
277 struct minmea_sentence_rmc *rmc = &(nmea_ptr->parsed_rmc);
278 if (!rmc->valid)
279 return false;
280
281 // nuova riga da aggiungere
282 char newline[96];
283 float lat = minmea_tocoord(&(rmc->latitude));
284 float lon = minmea_tocoord(&(rmc->longitude));
285 int newn = snprintf(newline, sizeof(newline), "%.6f,%.6f\n", lat, lon);
286 if (newn <= 0)
287 return false;
288
289 // buffer per massimo 20 righe
290 char lines[GPS_MAX_LINES][96] = {{0}};
291 int count = 0;
292
293 inPrinting = true;
294
295 // 1) leggi file esistente (se c'è)
296 FILE *f = fopen(GPS_LOG_PATH, "r");
297 if (f) {
298 char tmp[96];
299 while (fgets(tmp, sizeof(tmp), f)) {
300 // copia la riga, limitando dimensione
301 strncpy(lines[count % GPS_MAX_LINES], tmp, sizeof(lines[0]) - 1);
302 lines[count % GPS_MAX_LINES][sizeof(lines[0]) - 1] = '\0';
303 count++;
304 if (count > 200)
305 break; // safety (non dovrebbe mai succedere)
306 }
307 fclose(f);
308 }
309
310 // 2) costruisci nuova lista: ultime 19 + nuova
311 int start = (count > GPS_MAX_LINES - 1) ? (count - (GPS_MAX_LINES - 1)) : 0;
312 int outCount = 0;
313
314 char out[GPS_MAX_LINES][96] = {{0}};
315 for (int i = start; i < count; i++) {
316 strncpy(out[outCount], lines[i % GPS_MAX_LINES], sizeof(out[0]) - 1);
317 out[outCount][sizeof(out[0]) - 1] = '\0';
318 outCount++;
319 }
320
321 // aggiungi nuova riga
322 strncpy(out[outCount], newline, sizeof(out[0]) - 1);
323 out[outCount][sizeof(out[0]) - 1] = '\0';
324 outCount++;
325
326 // 3) riscrivi file con esattamente outCount righe (max 20)
327 f = fopen(GPS_LOG_PATH, "w");
328 if (!f) {
329 inPrinting = false;
330 return false;
331 }
332
333 for (int i = 0; i < outCount; i++) {
334 size_t n = strlen(out[i]);
335 if (n)
336 fwrite(out[i], 1, n, f);
337 }
338 fclose(f);
339
340 inPrinting = false;
341 return true;
342}

503void check_GPS_sync_req(queue_t *recv_queue, queue_t *send_queue,
504 struct parsed_nmea *nmea_ptr) {

506 uint8_t cmd;

508 if (cmd == GPS_SYNC_REQ) {
509 if (GPS_sync(nmea_ptr, true))
510 exit(EXIT_FAILURE);
511 cmd = GPS_SYNC_ACK;
512 if (queue_try_add(send_queue, &cmd))
513 ;
514 }
515 }
516 return;
517}
