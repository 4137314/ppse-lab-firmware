53bool nmea_gps_parse(String* nmea_message, struct parsed_nmea* nmea_ptr) {

55
56 if (nmea_ptr == NULL) {

78 case MINMEA_SENTENCE_GST: {
79 struct minmea_sentence_gst* frame = &(nmea_ptr->parsed_gst);
80 if (minmea_parse_gst(frame, nmea_message->c_str()))
81 print_NMEA_gst(frame);
82 else
83 (state = false);
84 } break;
85
86 case MINMEA_SENTENCE_GSV: {
87 struct minmea_sentence_gsv* frame = &(nmea_ptr->parsed_gsv);
88 if (minmea_parse_gsv(frame, nmea_message->c_str()))
89 print_NMEA_gsv(frame);

107 (state = false);

110 (state = false);

112 }

114 if (state)
115 return true;
116#if DEBUG
117 Serial.print("WARNING: Could not parse this nmea string ");
118 Serial.println(nmea_message->c_str());
119#endif
120 return false;

123bool gpsAcquire(enum minmea_sentence_id sentence_type, struct parsed_nmea* nmea_ptr) {
124 String nmea_message;
125
126 digitalWrite(LED_ALIVE, HIGH);
127
128 for (int i = 0; i < GPS_ACQUIRE_MAX_TRIES; ++i) {
129 for (int c = 0; Serial2.available() == 0; ++c) delay(1);
130
131 nmea_message = Serial2.readStringUntil('\n');
132 digitalWrite(LED_ALIVE, LOW);
133
134 if (minmea_sentence_id(nmea_message.c_str(), true) == sentence_type) {
135 if (nmea_gps_parse(&nmea_message, nmea_ptr))
136 return true;
137 }
138 }
139
140#if DEBUG
141 Serial.println(
142 "WARNING in gpsAcquire: max tries reached before correctly "
143 "acquiring the requested gps message\n");
144#endif
145 return false;
146}

168bool GPS_sync(struct parsed_nmea* nmea_ptr, bool force) {
169 // retain value across function calls
170 static unsigned long lastGPSsync = 0, lastGPSwrite = 0;
171
172 if ((millis() - lastGPSsync > GPS_SYNC_TIMEOUT_MSEC) || force) {
173 if (GetPosition_and_Satellites(nmea_ptr) && GetDate_and_Time(nmea_ptr)) {
174 lastGPSsync = millis();
175 if (((millis() - lastGPSwrite) > GPS_WRITE_FREQ) ||
176 force) // Write only at a certain frequency otherwise flash will wear
177 // out
178 {
179 if (save_gps_last(nmea_ptr))
180 lastGPSwrite = millis();
181 else
182 return false;
183 }
184 } else // It's only a warning so return true at the end
185 {
186#if DEBUG
187 Serial.println("WARNING in GPS_sync: failed to get data");
188#endif
189 }
190 }
191 return true;
192}

218bool save_gps_log20(struct parsed_nmea* nmea_ptr) {
219 if (inPrinting || driveConnected || updated)
220 return false;
221
222 struct minmea_sentence_rmc* rmc = &(nmea_ptr->parsed_rmc);
223 if (!rmc->valid)
224 return false;
225
226 // nuova riga da aggiungere
227 char newline[96];
228 float lat = minmea_tocoord(&(rmc->latitude));
229 float lon = minmea_tocoord(&(rmc->longitude));
230 int newn = snprintf(newline, sizeof(newline), "%.6f,%.6f\n", lat, lon);
231 if (newn <= 0)
232 return false;
233
234 // buffer per massimo 20 righe
235 char lines[GPS_MAX_LINES][96] = {{0}};
236 int count = 0;
237
238 inPrinting = true;
239
240 // 1) leggi file esistente (se c'è)
241 FILE* f = fopen(GPS_LOG_PATH, "r");
242 if (f) {
243 char tmp[96];
244 while (fgets(tmp, sizeof(tmp), f)) {
245 // copia la riga, limitando dimensione
246 strncpy(lines[count % GPS_MAX_LINES], tmp, sizeof(lines[0]) - 1);
247 lines[count % GPS_MAX_LINES][sizeof(lines[0]) - 1] = '\0';
248 count++;
249 if (count > 200)
250 break; // safety (non dovrebbe mai succedere)
251 }
252 fclose(f);
253 }
254
255 // 2) costruisci nuova lista: ultime 19 + nuova
256 int start = (count > GPS_MAX_LINES - 1) ? (count - (GPS_MAX_LINES - 1)) : 0;
257 int outCount = 0;
258
259 char out[GPS_MAX_LINES][96] = {{0}};
260 for (int i = start; i < count; i++) {
261 strncpy(out[outCount], lines[i % GPS_MAX_LINES], sizeof(out[0]) - 1);
262 out[outCount][sizeof(out[0]) - 1] = '\0';
263 outCount++;
264 }
265
266 // aggiungi nuova riga
267 strncpy(out[outCount], newline, sizeof(out[0]) - 1);
268 out[outCount][sizeof(out[0]) - 1] = '\0';
269 outCount++;
270
271 // 3) riscrivi file con esattamente outCount righe (max 20)
272 f = fopen(GPS_LOG_PATH, "w");
273 if (!f) {
274 inPrinting = false;
275 return false;
276 }
277
278 for (int i = 0; i < outCount; i++) {
279 size_t n = strlen(out[i]);
280 if (n)
281 fwrite(out[i], 1, n, f);
282 }
283 fclose(f);
284
285 inPrinting = false;
286 return true;
287}
