58void Weather_HandleLine(const String &line) {
59 // Parsing dati correnti
60 if (line.startsWith("WXC,")) {
61 int pos = 4;
62 String city, t, w, hum, code, wd;
63 if (!splitField(line, pos, city)) return;
64 if (!splitField(line, pos, t)) return;
65 if (!splitField(line, pos, w)) return;
66 if (!splitField(line, pos, hum)) return;
67 if (!splitField(line, pos, code)) return;
68 if (!splitField(line, pos, wd)) return;
69
70 city.replace(",", "_");
71 city.toCharArray(wx.city, sizeof(wx.city));
72 wx.curTempC = t.toFloat();
73 wx.curWindKmh = w.toFloat();
74 wx.curHumidity = hum.toInt();
75 wx.curWcode = code.toInt();
76 wx.weekday0 = (uint8_t)wd.toInt();
77 wx.curValid = true;
78 wx.lastUpdateMs = millis();
79 return;
80 }
81
82 // Parsing previsioni giornaliere
83 if (line.startsWith("WXD,")) {
84 int pos = 4;
85 String dStr;
86 if (!splitField(line, pos, dStr)) return;
87 int d = dStr.toInt();
88 if (d < 0 || d > 6) return;
89
90 // Estrazione 24 temperature orarie
91 for (int h = 0; h < 24; h++) {
92 String v;
93 if (!splitField(line, pos, v)) return;
94 wx.tempC[d][h] = v.toFloat();
95 }
96 // Estrazione 24 codici meteo orari (WMO codes)
97 for (int h = 0; h < 24; h++) {
98 String v;
99 if (!splitField(line, pos, v)) return;
100 int code = v.toInt();
101 wx.wcode[d][h] = (uint8_t)constrain(code, 0, 255);
102 }
103
104 // Aggiorna maschera bitwise per verificare se tutti i 7 giorni sono stati ricevuti
105 wx.recvMask |= (1u << d);
106 wx.fcValid = (wx.recvMask == 0x7F);
107 wx.lastUpdateMs = millis();
108 return;
109 }
110}
