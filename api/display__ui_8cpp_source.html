212 display.print(Weather_CodeToShortText(code));
213 } else {
214 display.print("Fc: no data");
215 }
216
217 // Riga 4: Now (corrente) + Humidita corrente
218 display.setCursor(0, 46);
219 display.print("Now: ");
220 if (wx.curValid) {
221 display.print(wx.curTempC, 1);
222 display.print("C ");
223 display.print(Weather_CodeToShortText(wx.curWcode));
224 } else {
225 display.print("NA");
226 }

228 display.setCursor(0, 56);
229 display.print("Hum: ");
230 // usa il NOME che hai realmente in WeatherState:
