383bool checkLeftRightCombo(KeyEvent ev) {
384 static uint32_t lastL = 0;
385 static uint32_t lastR = 0;
386
387 uint32_t now = millis();
388
389 if (ev == KEY_LEFT) lastL = now;
390 if (ev == KEY_RIGHT) lastR = now;
391
392 if (lastL && lastR && (uint32_t)abs((int32_t)(lastL - lastR)) <= COMBO_MS) {
393 lastL = 0;
394 lastR = 0;
395 onLeftRightCombo();
396 return true; // combo gestita
397 }
398
399 // pulizia (opzionale)
400 if (lastL && (now - lastL > COMBO_MS)) lastL = 0;
401 if (lastR && (now - lastR > COMBO_MS)) lastR = 0;
402
403 return false;
404}
