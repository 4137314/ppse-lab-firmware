44 if (events & GPIO_IRQ_EDGE_RISE) {
45 gpio_set_irq_enabled(gpio, GPIO_IRQ_EDGE_RISE, false);
46 gpio_set_irq_enabled(gpio, GPIO_IRQ_EDGE_FALL, true);
47 return;

340bool checkLeftRightCombo(KeyEvent ev) {
341 static uint32_t lastL = 0;
342 static uint32_t lastR = 0;
343
344 uint32_t now = millis();
345
346 if (ev == KEY_LEFT)
347 lastL = now;
348 if (ev == KEY_RIGHT)
349 lastR = now;
350
351 if (lastL && lastR && (uint32_t)abs((int32_t)(lastL - lastR)) <= COMBO_MS) {
352 lastL = 0;
353 lastR = 0;
354 onLeftRightCombo();
355 return true; // combo gestita
356 }
357
358 // pulizia (opzionale)
359 if (lastL && (now - lastL > COMBO_MS))
360 lastL = 0;
361 if (lastR && (now - lastR > COMBO_MS))
362 lastR = 0;
363
364 return false;
365}
